WRITE "Ingresa la primera calificacion:";
READ a;

WRITE "Ingresa la segunda calificacion:";
READ b;

WRITE "Ingresa la tercera calificacion:";
READ c;

SET promedio TO (a + b + c) / 3;

WRITE "El promedio es:";
WRITE promedio;

IF promedio >= 70 THEN
    WRITE "Aprobado";
ELSE
    WRITE "Reprobado";
ENDIF
