// --- DECLARACIONES DE MATRICES Y VARIABLES ---
NEW MATRIX CALIFICACIONES[3][5];        // Calificaciones: 3 alumnos, 5 materias
NEW MATRIX PROMEDIOS_GENERALES[3][1];   // Promedio general de cada alumno
NEW MATRIX ALUMNOS_ORDENADOS[3][1];     // Almacena los índices de los alumnos ordenados por promedio
NEW MATRIX PROMEDIOS_TEMPORALES[3][1];  // Para el proceso de ordenamiento

// Variables individuales para almacenar los nombres de los alumnos
SET NOMBRE_ALUMNO_0 TO " "; // Inicializar con cadena vacía o similar si tu lenguaje lo permite
SET NOMBRE_ALUMNO_1 TO " ";
SET NOMBRE_ALUMNO_2 TO " ";

// Variables para el ciclo y cálculos
SET SUMA_MATERIAS TO 0;
SET I_ALUMNO TO 0;
SET I_MATERIA TO 0;
SET J_COMPARE TO 0;
SET TEMP_PROMEDIO TO 0;
SET TEMP_ALUMNO_INDICE TO 0;
SET RANK_IDX TO 0;
SET PROMEDIO_VAL TO 0;

WRITE "--- INGRESO DE DATOS ---";

// --- CICLO PARA LEER NOMBRES Y CALIFICACIONES DE CADA ALUMNO ---
FOR I_ALUMNO = 0 TO 2 // Itera 3 veces (para Alumno 0, Alumno 1, Alumno 2)

    WRITE "Ingresa el nombre para el Alumno ";
    WRITE I_ALUMNO;
    WRITE ":";

    // Leer el nombre del alumno y asignarlo a la variable correcta
    IF I_ALUMNO EQ 0 THEN READ NOMBRE_ALUMNO_0; ENDIF
    IF I_ALUMNO EQ 1 THEN READ NOMBRE_ALUMNO_1; ENDIF
    IF I_ALUMNO EQ 2 THEN READ NOMBRE_ALUMNO_2; ENDIF

    WRITE "Ingresa calificaciones para este alumno (5 materias):";

    FOR I_MATERIA = 0 TO 4 // Itera 5 veces para las materias
        WRITE "  Materia ";
        WRITE I_MATERIA;
        WRITE ":";
        READ CALIFICACIONES[I_ALUMNO][I_MATERIA];
    ENDFOR
ENDFOR

WRITE "--- CALCULANDO PROMEDIOS ---";

// --- CALCULAR PROMEDIOS GENERALES ---
FOR I_ALUMNO = 0 TO 2
    SET SUMA_MATERIAS TO 0;
    FOR I_MATERIA = 0 TO 4
        SET SUMA_MATERIAS TO SUMA_MATERIAS + CALIFICACIONES[I_ALUMNO][I_MATERIA];
    ENDFOR
    SET PROMEDIOS_GENERALES[I_ALUMNO][0] TO SUMA_MATERIAS / 5; // Asume 5 materias fijas
ENDFOR

WRITE "--- IMPRIMIENDO BOLETAS INDIVIDUALES ---";

// --- IMPRIMIR BOLETA PARA CADA ALUMNO ---
FOR I_ALUMNO = 0 TO 2
    WRITE " "; // Salto de línea para separar boletas
    WRITE "BOLETA DE CALIFICACIONES - ";

    // Imprimir el nombre del alumno actual
    IF I_ALUMNO EQ 0 THEN WRITE NOMBRE_ALUMNO_0; ENDIF
    IF I_ALUMNO EQ 1 THEN WRITE NOMBRE_ALUMNO_1; ENDIF
    IF I_ALUMNO EQ 2 THEN WRITE NOMBRE_ALUMNO_2; ENDIF

    WRITE "Calificaciones por Materia:";
    FOR I_MATERIA = 0 TO 4
        WRITE "  Materia ";
        WRITE I_MATERIA; // Aún no podemos imprimir nombres de materias
        WRITE ": ";
        WRITE CALIFICACIONES[I_ALUMNO][I_MATERIA];
    ENDFOR

    WRITE "PROMEDIO GENERAL: ";
    WRITE PROMEDIOS_GENERALES[I_ALUMNO][0];
    WRITE " "; // Otro salto de línea
ENDFOR


WRITE "--- ORDENANDO ALUMNOS POR APROVECHAMIENTO ---";

// --- PREPARAR PARA ORDENAMIENTO (COPIAR PROMEDIOS E INICIALIZAR INDICES) ---
SET ALUMNOS_ORDENADOS[0][0] TO 0;
SET ALUMNOS_ORDENADOS[1][0] TO 1;
SET ALUMNOS_ORDENADOS[2][0] TO 2;

SET PROMEDIOS_TEMPORALES[0][0] TO PROMEDIOS_GENERALES[0][0];
SET PROMEDIOS_TEMPORALES[1][0] TO PROMEDIOS_GENERALES[1][0];
SET PROMEDIOS_TEMPORALES[2][0] TO PROMEDIOS_GENERALES[2][0];

// --- ALGORITMO DE ORDENAMIENTO BURBUJA (DE MAYOR A MENOR PROMEDIO) ---
// (Para 3 elementos, 2 pasadas son suficientes)
// Pasada 1
FOR I_ALUMNO = 0 TO 1
    FOR J_COMPARE = 0 TO 1
        IF PROMEDIOS_TEMPORALES[J_COMPARE][0] LT PROMEDIOS_TEMPORALES[J_COMPARE+1][0] THEN
            SET TEMP_PROMEDIO TO PROMEDIOS_TEMPORALES[J_COMPARE][0];
            SET PROMEDIOS_TEMPORALES[J_COMPARE][0] TO PROMEDIOS_TEMPORALES[J_COMPARE+1][0];
            SET PROMEDIOS_TEMPORALES[J_COMPARE+1][0] TO TEMP_PROMEDIO;

            SET TEMP_ALUMNO_INDICE TO ALUMNOS_ORDENADOS[J_COMPARE][0];
            SET ALUMNOS_ORDENADOS[J_COMPARE][0] TO ALUMNOS_ORDENADOS[J_COMPARE+1][0];
            SET ALUMNOS_ORDENADOS[J_COMPARE+1][0] TO TEMP_ALUMNO_INDICE;
        ENDIF
    ENDFOR
ENDFOR

// Pasada 2
FOR J_COMPARE = 0 TO 1
    IF PROMEDIOS_TEMPORALES[J_COMPARE][0] LT PROMEDIOS_TEMPORALES[J_COMPARE+1][0] THEN
        SET TEMP_PROMEDIO TO PROMEDIOS_TEMPORALES[J_COMPARE][0];
        SET PROMEDIOS_TEMPORALES[J_COMPARE][0] TO PROMEDIOS_TEMPORALES[J_COMPARE+1][0];
        SET PROMEDIOS_TEMPORALES[J_COMPARE+1][0] TO TEMP_PROMEDIO;

        SET TEMP_ALUMNO_INDICE TO ALUMNOS_ORDENADOS[J_COMPARE][0];
        SET ALUMNOS_ORDENADOS[J_COMPARE][0] TO ALUMNOS_ORDENADOS[J_COMPARE+1][0];
        SET ALUMNOS_ORDENADOS[J_COMPARE+1][0] TO TEMP_ALUMNO_INDICE;
    ENDIF
ENDFOR


WRITE "--- RANKING DE APROVECHAMIENTO FINAL ---";

// --- IMPRIMIR PRIMER LUGAR ---
SET RANK_IDX TO ALUMNOS_ORDENADOS[0][0];
SET PROMEDIO_VAL TO PROMEDIOS_GENERALES[RANK_IDX][0];
WRITE "PRIMER LUGAR: ";
IF RANK_IDX EQ 0 THEN WRITE NOMBRE_ALUMNO_0; ENDIF
IF RANK_IDX EQ 1 THEN WRITE NOMBRE_ALUMNO_1; ENDIF
IF RANK_IDX EQ 2 THEN WRITE NOMBRE_ALUMNO_2; ENDIF
WRITE " con promedio: ";
WRITE PROMEDIO_VAL;

// --- IMPRIMIR SEGUNDO LUGAR ---
SET RANK_IDX TO ALUMNOS_ORDENADOS[1][0];
SET PROMEDIO_VAL TO PROMEDIOS_GENERALES[RANK_IDX][0];
WRITE "SEGUNDO LUGAR: ";
IF RANK_IDX EQ 0 THEN WRITE NOMBRE_ALUMNO_0; ENDIF
IF RANK_IDX EQ 1 THEN WRITE NOMBRE_ALUMNO_1; ENDIF
IF RANK_IDX EQ 2 THEN WRITE NOMBRE_ALUMNO_2; ENDIF
WRITE " con promedio: ";
WRITE PROMEDIO_VAL;

// --- IMPRIMIR TERCER LUGAR ---
SET RANK_IDX TO ALUMNOS_ORDENADOS[2][0];
SET PROMEDIO_VAL TO PROMEDIOS_GENERALES[RANK_IDX][0];
WRITE "TERCER LUGAR: ";
IF RANK_IDX EQ 0 THEN WRITE NOMBRE_ALUMNO_0; ENDIF
IF RANK_IDX EQ 1 THEN WRITE NOMBRE_ALUMNO_1; ENDIF
IF RANK_IDX EQ 2 THEN WRITE NOMBRE_ALUMNO_2; ENDIF
WRITE " con promedio: ";
WRITE PROMEDIO_VAL;
